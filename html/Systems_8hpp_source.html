Rtype
R-Type is a side-scrolling shoot 'em up arcade game developed by Irem and released in 1987. It is the first installment in the R-Type series. Players control the R-9 Arrowhead spacecraft, aiming to defeat the Bydo, an alien race seeking to eradicate humanity. The R-9 can acquire a device called Force, which provides protection against enemy attacks and enhances firepower. In North America, the arcade version was distributed by Nintendo, marking the last arcade game they distributed.

125template <typename T> Position *get_nearest_player_position(Position &monsterPosition, Sparse_array<T> &players, Sparse_array<Position> &positions)
126{
127 Position *nearestPlayerPosition = nullptr;
128 float minDistance = 4852800;
129
130 for (size_t i = 0; i < players.size(); i++) {
131 if (players[i] && positions[i]) {
132 auto &playerPosition = positions[i].value();
133
134 // Calculate the Euclidean distance
135 float dx = playerPosition.x - monsterPosition.x;
136 float dy = playerPosition.y - monsterPosition.y;
137 float distanceSquared = dx * dx + dy * dy;
138
139 // Check if this player is closer
140 if (distanceSquared < minDistance) {
141 minDistance = distanceSquared;
142 nearestPlayerPosition = &positions[i].value();
143 }
144 }
145 }
146
147 return nearestPlayerPosition;
148}
